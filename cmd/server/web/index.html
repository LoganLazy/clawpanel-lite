    async function bootInit() {
      try {
        const status = await jsonFetch('/api/status');
        const installed = !(status.openclaw || '').includes('not found');
        if (!installed) {
          document.getElementById('initOverlay').classList.remove('hidden');
        }
      } catch (e) {}
    }

    bootInit();

    function selectGwBind(bind) {
      document.getElementById('gwBind').value = bind;
      document.querySelectorAll('.gw-option-card').forEach(card => {
        card.classList.toggle('selected', card.dataset.bind === bind);
      });
    }
    function toggleGwAdvanced() {
      const panel = document.getElementById('gwAdvanced');
      if (!panel) return;
      panel.style.display = (panel.style.display === 'none') ? 'block' : 'none';
    }
